export default {
  data(){
    return{
        //多项选择
      dialogVisible:false,
      multipleData:[
        {
          id:1,
          name:'测试001',
        },{
          id:2,
          name:'测试002',
        },{
          id:3,
          name:'测试003',
        }
      ],
      selectedList:[],
    }
  },
  mounted(){

  },
  methods:{
    confirm(item){
      this.selectedList = item.map(arr =>{return arr.id })
      this.dialogVisible = false
    },
    cancel(){
      this.dialogVisible = false
    },
  }
